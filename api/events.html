# change

  • type: Boolean

When the state changes, it returns the boolean value.

<template>
  <div id="app">
    <VueFixedHeader @change="handleChangeFixedHeader">
      <nav>
        <ul>
          <li>

          </li>
        </ul>
      </nav>
    </VueFixedHeader>
  </div>
</template>

<script>
import Vue from 'vue'

    VueFixedHeader

      console.log(isFixed)
    }
  }
})
</script>
